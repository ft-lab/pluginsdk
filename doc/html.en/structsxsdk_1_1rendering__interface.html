 void create_animation (scene_interface* scene, const char* filename, int start, int end) { 
   compointer<rendering_interface> render(scene->get_rendering_interface()); 
   render->start_animation(filename); 
   for(int frame = start ; frame >= end ; ++frame) { 
     render->render(); 
     render->append_animation(); 
   } 
   render->finish_animation(); 
 }
